    Info

    Given an age in seconds, calculate how old someone would be on:

    • Earth: orbital period 365.25 Earth days, or 31557600 seconds
    • Mercury: orbital period 0.2408467 Earth years
    • Venus: orbital period 0.61519726 Earth years
    • Mars: orbital period 1.8808158 Earth years
    • Jupiter: orbital period 11.862615 Earth years
    • Saturn: orbital period 29.447498 Earth years
    • Uranus: orbital period 84.016846 Earth years
    • Neptune: orbital period 164.79132 Earth years

    space_age.cpp

    #include "space_age.h"

            #include <string>
            #include <iostream>

            namespace space_age {

  // const double space_age::s_earth = 1;
  const double space_age::s_mercury = 0.2408467;
  const double space_age::s_venus = 0.61519726;
  const double space_age::s_mars = 1.8808158;
  const double space_age::s_jupiter = 11.862615;
  const double space_age::s_saturn = 29.447498;
  const double space_age::s_uranus = 84.016846;
  const double space_age::s_neptune = 164.79132;
 const double space_age::s_to_years = 60 * 60 * 24 * 365.25;
    

              space_age::space_age(double earth_seconds) {
                m_age = earth_seconds;
              }

              double space_age::seconds() const {
                return m_age;
              }

              double space_age::on_earth() const {
                return seconds() / s_to_years;
              }

              double space_age::on_mercury() const {
                return on_earth() / s_mercury;
              }

              double space_age::on_venus() const {
                return on_earth() / s_venus;
              }

              double space_age::on_mars() const {
                return on_earth() / s_mars;
              }

              double space_age::on_jupiter() const {
                return on_earth() / s_jupiter;
              }

              double space_age::on_saturn() const {
                return on_earth() / s_saturn;
              }

              double space_age::on_uranus() const {
                return on_earth() / s_uranus;
              }

              double space_age::on_neptune() const {
                return on_earth() / s_neptune;
              }
