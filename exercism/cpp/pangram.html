    Info

    Determine if a sentence is a pangram. A pangram (Greek: παν γράμμα, pan gramma, "every letter") is a sentence using every letter of the alphabet at least once. The best known English pangram is: The quick brown fox jumps over the lazy dog. The alphabet used consists of ASCII letters a to z, inclusive, and is case insensitive. Input will not contain non-ASCII symbols.

    pangram.h

    #ifndef PANGRAM_H
#define PANGRAM_H

#include <string>
#include <vector>

namespace pangram {
  static const unsigned int alphabet_size = 26;
  bool is_pangram(std::string pangram);
}

#endif

    pangram.cpp

    #include "pangram.h"

#include <string>
#include <vector>
#include <algorithm> //std::transform()
#include <numeric> //std::iota()

namespace pangram {
  bool is_pangram(std::string pangram) {
    std::vector<char> alphabet(alphabet_size);
    std::iota(alphabet.begin(), alphabet.end(), 'a');
    std::transform(pangram.begin(), pangram.end(), pangram.begin(), ::tolower);

    for (auto letter : pangram)
      alphabet.erase(std::remove(alphabet.begin(), alphabet.end(), letter), alphabet.end());

    return alphabet.empty();
  }

}
