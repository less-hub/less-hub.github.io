    Info

    Calculate the number of grains of wheat on a chessboard given that the number on each square doubles. There once was a wise servant who saved the life of a prince. The king promised to pay whatever the servant could dream up. Knowing that the king loved chess, the servant told the king he would like to have grains of wheat. One grain on the first square of a chess board, with the number of grains doubling on each successive square. There are 64 squares on a chessboard (where square 1 has one grain, square 2 has two grains, and so on). Write code that shows: how many grains were on a given square, and the total number of grains on the chessboard

    grains.h

    #ifndef GRAINS_H
            #define GRAINS_H

            #include <map>

            namespace grains {
              const unsigned int max_number_of_squares = 64;
              const unsigned int grain_growing_rate = 2;
              static std::map<int, unsigned long long> stored_grain;

              unsigned long long square(const unsigned int chessboard_position);
              unsigned long long total();

              std::map<int, unsigned long long>::iterator find_highest_stored_grain();
              void throw_if_invalid_chessboard_position(const unsigned int chessboard_position);
            }

            #endif

    grains.cpp

    #include "grains.h"

#include <stdexcept>
#include <map>
#include <string>
#include <algorithm> //std::max_element()

namespace grains {
  unsigned long long square (const unsigned int chessboard_position) {
    throw_if_invalid_chessboard_position(chessboard_position);
    std::map<int, unsigned long long>::iterator grain = stored_grain.find(chessboard_position);

    if (stored_grain.empty())
      stored_grain.emplace(1, 1);

    if (grain != stored_grain.end())
      return grain->second;

    grain = find_highest_stored_grain();

    unsigned int chessboard_position_difference = chessboard_position - grain->first;
    unsigned long long grain_at_chessboard_position = grain->second;

    for (unsigned int pos = 0; pos < chessboard_position_difference; pos++) {
      grain_at_chessboard_position *= grain_growing_rate;
      stored_grain.emplace((grain->first + pos + 1), grain_at_chessboard_position);
    }

    return grain_at_chessboard_position;
  }

  unsigned long long total () {
    unsigned long long total_grain_in_chessboard = 0;

    for (unsigned int pos = 1; pos <= max_number_of_squares; pos++)
      total_grain_in_chessboard += square(pos);

    return total_grain_in_chessboard;
  }

  std::map<int, unsigned long long>::iterator find_highest_stored_grain() {
    return std::max_element(stored_grain.begin(), stored_grain.end(), stored_grain.value_comp());
  }

  void throw_if_invalid_chessboard_position( const unsigned int chessboard_position ) {
    if (chessboard_position > max_number_of_squares)
      throw std::domain_error("Domain error: The position is out of range!");
  }
}
